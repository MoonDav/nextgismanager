206 static bool CreateAndRunThread(wxThread* const pThread, wxString sClassName = wxEmptyString, const wxString sThreadName = wxEmptyString, int nPriority = WXTHREAD_DEFAULT_PRIORITY)
207 {
208  wxCHECK_MSG(pThread, false, wxT("Input wxThread pointer is null"));
209 
210  if(sClassName.IsEmpty())
211  sClassName = wxString(_("wxGISCore"));
212  if ( pThread->Create() != wxTHREAD_NO_ERROR )
213  {
214  wxLogError(_("%s: Can't create %s Thread!"), sClassName.c_str(), sThreadName.c_str());
215  return false;
216  }
217  pThread->SetPriority(nPriority);
218  if(pThread->Run() != wxTHREAD_NO_ERROR )
219  {
220  wxLogError(_("%s: Can't run %s Thread!"), sClassName.c_str(), sThreadName.c_str());
221  return false;
222  }
223  return true;
224 }
